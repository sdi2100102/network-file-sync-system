create socket

    int listening_sock;

    if ((listening_sock = socket(AF_INET, SOCK_STREAM, 0)) < 0)
        perror_exit("socket");

bind socket to port

    struct sockaddr_in server;
    int port;

    struct sockaddr *serverptr=(struct sockaddr *)&server;
    server.sin_family = AF_INET;       /* Internet domain */
    server.sin_addr.s_addr = htonl(INADDR_ANY);
    server.sin_port = htons(port);   
    /* Bind socket to address */
    if (bind(listening_sock, serverptr, sizeof(server)) < 0)
        perror_exit("bind");

listen

    if (listen(listening_sock, 5) < 0) perror_exit("listen");

accept

    int sock_fd;
    if ((sock_fd = accept(listening_sock, clientptr, &clientlen)) < 0) perror_exit("accept");


close

    close(listening_sock);
    close(sock_fd);
